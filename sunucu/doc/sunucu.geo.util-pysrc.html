 55    while aggregate_reverse_index > 0: 
 56      pull_arr_index = None 
 57      pull_val = None 
 58   
 59      for i in range(len(lists)): 
 60        if reverse_indices[i] == 0: 
 61          # Reached the end of this list. 
 62          pass

 65          # Found a duplicate, advance the index of the list in which the 
 66          # duplicate was found. 
 67          reverse_indices[i] -= 1 
 68          aggregate_reverse_index -= 1 
 69        elif (pull_arr_index is None or
